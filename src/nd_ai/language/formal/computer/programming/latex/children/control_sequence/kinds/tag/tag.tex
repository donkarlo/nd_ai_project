\usepackage{xcolor}
\usepackage[most]{tcolorbox}
\usepackage{xparse}

\definecolor{TagBackground}{HTML}{DCEEFF}
\definecolor{TagText}{HTML}{1E5FAF}

\newtcbox{\SemanticTagBox}{
    on line,
    tcbox raise base,
    boxrule=0pt,
    arc=2pt,
    boxsep=0pt,
    left=5pt,
    right=5pt,
    top=1pt,
    bottom=1pt,
    colback=TagBackground,
    coltext=TagText,
    fontupper=\sffamily\footnotesize,
    nobeforeafter
}

\ExplSyntaxOn

\cs_new_protected:Npn \semantic_tags_print:n #1
    {
    \clist_map_inline:nn {#1}
    {
        \SemanticTagBox{\strut\tl_trim_spaces:n {##1}}
        \hspace{0.35em}
    }
}

\cs_if_exist:NTF \tag
{
    \RenewDocumentCommand{\tag}{m}
    {
        \semantic_tags_print:n {#1}
    }
}
{
    \NewDocumentCommand{\tag}{m}
    {
        \semantic_tags_print:n {#1}
    }
}

\ExplSyntaxOff
